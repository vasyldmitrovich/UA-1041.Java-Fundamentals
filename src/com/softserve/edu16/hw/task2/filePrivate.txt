private class Student {
    private String name;
    private String group;

    private int course;

    private List<Integer> grades;

    private Student(String name, String group, int course, List<Integer> grades) {
        this.name = name;
        this.group = group;
        this.course = course;
        this.grades = grades;
    }

    private String getName() {
        return name;
    }

    private void setName(String name) {
        this.name = name;
    }

    private String getGroup() {
        return group;
    }

    private void setGroup(String group) {
        this.group = group;
    }

    private int getCourse() {
        return course;
    }

    private void setCourse(int course) {
        this.course = course;
    }

    private List<Integer> getGrades() {
        return grades;
    }

    private void setGrades(List<Integer> grades) {
        this.grades = grades;
    }

    @Override
    private String toString() {
        return "Student{" +
                "name='" + name + '\'' +
                ", group='" + group + '\'' +
                ", course=" + course +
                '}';
    }
}
